function [MuaT] = funPR4(dominio)
[filas, columnas]=size(dominio);
%Función de membresía asiganada Mua inciso a) de la imagen 4 (OperDif4)
for i=1:1:columnas
if (dominio(i) <= 1.7)
MuaT(1,i)=0;
end
if (1.7 <= dominio(i) && dominio(i) <= 1)
MuaT(1,i)=(dominio(i)-1.7)/(2-1.7);
end
if(2 <= dominio(i) && dominio(i) <=2.2)
MuaT(1,i)=(2.2-dominio(i));
end
if(2.2 <= dominio(i))
MuaT(1,i)=0;
end
end
end
function [COM] = PR4_7(dominio,rango,complemento)
%Función para efectuar los tres tipos de complementos con conjuntos difusos
switch complemento
case 1
%Estandar
COM=1-rango;
case 2
%Sugeno
s=.1;
COM = (1-rango)./(1+s.*rango);
case 3
%Yager
w=.2;
COM=(1-rango.^w).^(1/w);
otherwise
disp('No hay función para este caso')
end
end

Código de la función del Problema 6

function PR6 (x,u,v)
tcon=1;
%GENERAMOS CONJUNTOS
A=trimf(x,u);
B=trimf(x,v);
%MAXIMO
M=max(A,B);
ttc(1)="Gráfica de Máximo";
%SUMA ALGEBRAICA
SAL=A+B-(A.*B);
ttc(2)="Gráfica de Suma Algebraica";
%SUMA ACOTADA
SAC=min(1,A+B);
ttc(3)="Gráfica de Suma Acotada";
%GRFICAMOS CONJUNTOS
for op=1:3
figure (op)
subplot(3,1,1)
GRAFICAR(tcon,x,A,[20/255 118/255 201/255],'Gráfica del conjunto difuso A')
subplot(3,1,2)
GRAFICAR(tcon,x,B,[255/255 190/255 51/255],'Gráfica del conjunto difuso B')
switch op
case 1
CC=M;
case 2
CC=SAL;
case 3
CC=SAC;
end
subplot(3,1,3)
GRAFICAR(tcon,x,CC,[111/255 201/255 20/255],ttc(op))
end
%GRAFICAMOS CONJUNTOS DE OPERACIONES
figure (op+1)
subplot(3,1,1)
GRAFICAR(tcon,x,M,[111/255 201/255 20/255],ttc(1))
subplot(3,1,2)
GRAFICAR(tcon,x,SAL,[111/255 201/255 20/255],ttc(2))
subplot(3,1,3)
GRAFICAR(tcon,x,SAC,[111/255 201/255 20/255],ttc(3))
end

Cálculos Matemáticos

function [MF] = PR10(NFM,x,k,a,m)
%Función para calculo de funciones de membresía solicitadas en problema 10
clc
close all
switch NFM
case 1
MF=1./(1+k*(x-m).^2);
case 2
[filas, columnas]=size(x);
for i=1:1:columnas
if (x(i) <= a)
MF(1,i)=0;
end
if (x(i) > a)
MF(1,i)=(k*(x(i)-a).^2)/(1+(k*(x(i)-a).^2));
end
end
case 3
[filas, columnas]=size(x);
for i=1:1:columnas
if (x(i) ~= m)
MF(1,i)=0;
end
if (x(i) == m)
MF(1,i)=1;
end
end
otherwise
disp('No hay función para este caso')
end
clc
end

function GRAFICAR(tcon,x,con,color,t)
%Graficar conjunto continuo
if (tcon==1)
plot(x,con,'Color',color)
%Graficar conjunto discreto
elseif(tcon==2)
stem(x,con,'Color',color)
end
%resto de informacion
grid on;
title(t)
h = gca; % ejes actuales
h.XAxisLocation = "origin"; % eje en el origen
h.YAxisLocation = "origin"; % eje en el origen
h.Box = "off";
end
function [] = PlotAndHold(dominio,rango1,rango2,tipo)
%función para graficar conjuntos
% tipo 0= dos conjuntos continuos en una misma gráfica
% tipo 1= un conjunto discreto
% tipo 2= un conjunto continuo
if (tipo == 0)
plot(dominio,rango1,'g','LineWidth',2);
grid on
axis([1.9,2.3,-.1,1.1])
hold on
plot(dominio,rango2,'r','LineWidth',2,'LineStyle',' : ');
elseif (tipo == 1)
stem(dominio,rango1,'Color','#7E2F8E','LineWidth',3);
grid on
axis([0,3.5,-0.1,1.1])
elseif(tipo == 2)
plot(dominio,rango1,'Color','#7E2F8E','LineWidth',2);
grid on
axis([-5,7,-0.1,1.1])
else
disp('valor no válido')
end
end
